    unlist()

# fixed width dataframe
ssamatab1 <- read_delim("./dataset/ssamatab1.txt",
                                                delim = "[ ]",
                                                col_names = "raw",
                                                skip = 5) %>%
    filter(str_detect(raw, "MT[0-9]+[\\s]")) %>%
    separate(raw, into = ssamatab1_colname, sep = "[\\s]{2,}") %>% 
    janitor::clean_names()
